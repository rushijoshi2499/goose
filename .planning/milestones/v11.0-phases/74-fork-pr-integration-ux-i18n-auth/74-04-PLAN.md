    CHANGE 9 — REPLACE pollForAuthorizationCode with new pollForToken that uses standard grant type and decodes authorization_pending from JSON body (not HTTP status code):

      private func pollForToken(_ deviceCode: CodexSelfContainedDeviceCode) async throws -> CodexTokenExchangeResponse {
        let startedAt = Date()
        while Date().timeIntervalSince(startedAt) < TimeInterval(maxDeviceCodeWaitSeconds) {
          try Task.checkCancellation()
          guard let url = URL(string: "\(issuer)/oauth/token") else {
            throw CodexSelfContainedAuthError.invalidURL("\(issuer)/oauth/token")
          }
          var request = URLRequest(url: url)
          request.httpMethod = "POST"
          request.setValue("application/x-www-form-urlencoded", forHTTPHeaderField: "Content-Type")
          request.setValue("application/json", forHTTPHeaderField: "Accept")
          request.setValue("goose-swift", forHTTPHeaderField: "originator")
          request.httpBody = formURLEncoded([
            "grant_type": "urn:ietf:params:oauth:grant-type:device_code",
            "device_code": deviceCode.loginID,
            "client_id": clientID,
          ])
          let (data, response) = try await session.data(for: request)
          // Standard RFC 8628: pending returns HTTP 200 with error JSON body (not 4xx)
          if let httpResponse = response as? HTTPURLResponse, (200..<300).contains(httpResponse.statusCode) {
            // Try to decode as error first — authorization_pending means keep waiting
            if let errorBody = try? JSONDecoder().decode(CodexDevicePollErrorResponse.self, from: data),
               let errorCode = errorBody.error {
              if errorCode == "authorization_pending" || errorCode == "slow_down" {
                let waitInterval = errorCode == "slow_down"
                  ? max(deviceCode.interval + 5, 1)
                  : max(deviceCode.interval, 1)
                try await Task.sleep(for: .seconds(waitInterval))
                continue
              }
              throw CodexSelfContainedAuthError.invalidResponse("Auth poll error: \(errorCode)")
            }
            // No error field — decode as successful token response
            return try JSONDecoder().decode(CodexTokenExchangeResponse.self, from: data)
          } else {
            try validate(response: response, data: data)  // throws httpStatus error
          }
        }
        throw CodexSelfContainedAuthError.invalidResponse("Device code login timed out after \(maxDeviceCodeWaitSeconds / 60) minutes.")
      }
