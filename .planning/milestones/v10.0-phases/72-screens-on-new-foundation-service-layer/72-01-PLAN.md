Edit store.rs — add `query_metric_series_range` after `insert_metric_series` (line ~7038):
```
pub fn query_metric_series_range(
    &self,
    metric_name: &str,
    start_date: &str,
    end_date: &str,
    source: Option<&str>,
) -> GooseResult<Vec<serde_json::Value>> {
    let conn = self.conn.lock().unwrap();
    let rows: Vec<serde_json::Value> = if let Some(src) = source {
        let mut stmt = conn.prepare(
            "SELECT date, value FROM metric_series \
             WHERE metric_name = ?1 AND source = ?2 \
               AND date >= ?3 AND date <= ?4 \
             ORDER BY date ASC",
        )?;
        stmt.query_map(params![metric_name, src, start_date, end_date], |row| {
            Ok((row.get::<_, String>(0)?, row.get::<_, f64>(1)?))
        })?
        .filter_map(|r| r.ok())
        .map(|(date, value)| json!({"date": date, "value": value}))
        .collect()
    } else {
        let mut stmt = conn.prepare(
            "SELECT date, value FROM metric_series \
             WHERE metric_name = ?1 AND date >= ?2 AND date <= ?3 \
             ORDER BY date ASC",
        )?;
        stmt.query_map(params![metric_name, start_date, end_date], |row| {
            Ok((row.get::<_, String>(0)?, row.get::<_, f64>(1)?))
        })?
        .filter_map(|r| r.ok())
        .map(|(date, value)| json!({"date": date, "value": value}))
        .collect()
    };
    Ok(rows)
}
```

Do NOT use string interpolation in SQL queries — use only rusqlite `params![]` macro. The `conn` lock is already the pattern used by `insert_metric_series` — follow it exactly.

fn metric_series_query_range_round_trip() {
    let tmp = tempfile::NamedTempFile::new().unwrap();
    let db_path = tmp.path().to_str().unwrap();
    // seed via upsert
    let upsert_req = json!({
        "schema": "goose.bridge.request.v1",

        "args": {
            "database_path": db_path,
            "source": "test",
            "metric_name": "recovery",
            "date": "2026-06-01",
            "value": 72.5
        }
    });
    let _ = handle_bridge_request(upsert_req.to_string());
    // query
    let query_req = json!({
        "schema": "goose.bridge.request.v1",
        "method": "metric_series.query_range",
        "args": {
            "database_path": db_path,
            "metric_name": "recovery",
            "start_date": "2026-06-01",
            "end_date": "2026-06-07"
        }
    });
    let resp: serde_json::Value = serde_json::from_str(&handle_bridge_request(query_req.to_string())).unwrap();
    assert!(resp["ok"].as_bool().unwrap_or(false), "query_range should succeed");
    let rows = resp["result"]["rows"].as_array().unwrap();
    assert_eq!(rows.len(), 1);
    assert_eq!(rows[0]["date"], "2026-06-01");
    assert!((rows[0]["value"].as_f64().unwrap() - 72.5).abs() < 0.001);
}
