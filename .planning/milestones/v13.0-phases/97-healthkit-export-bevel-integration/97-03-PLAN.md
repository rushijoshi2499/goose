    #if canImport(HealthKit)
    import Foundation
    import HealthKit

    extension GooseAppModel {
      // Called by MoreView toggle onChange when newValue == true (D-09).
      // Requests HK write authorization. On denial: sets toggle back off, logs error (D-07).
      func enableHealthKitExport() async {
        do {
          try await GooseHealthKitExporter.requestAuthorization()
          // Authorization succeeded or was already granted — toggle stays on.
          // Check actual authorization status for heartRate as a proxy:
          let hkStore = HKHealthStore()
          let status = hkStore.authorizationStatus(for: HKQuantityType(.heartRate))
          if status != .sharingAuthorized {
            // User denied — revert toggle (D-07).
            await MainActor.run {
              UserDefaults.standard.set(false, forKey: GooseHealthKitExporter.exportEnabledKey)
            }
            ble.record(level: .error, source: "healthkit", title: "HK export disabled", body: "Authorization denied by user")
          }
        } catch {
          // requestAuthorization threw — revert toggle (D-07).
          await MainActor.run {
            UserDefaults.standard.set(false, forKey: GooseHealthKitExporter.exportEnabledKey)
          }
          ble.record(level: .error, source: "healthkit", title: "HK auth error", body: "\(error)")
        }
      }
    }
    #endif

    Note: if HKHealthStore.requestAuthorization never throws on denial (it completes without error, just sets status to .sharingDenied), check the actual behavior against HealthKitFullImporter.swift's handling pattern. Adjust: if the continuation only errors on system-level failures, use authorizationStatus after the continuation completes to detect denial.
